let brokers: BrokerCustomerItem[] = [];
let brokerLoadError = "";

try {
  const brokerResult = await requestJson<{
    success: true;
    brokers: BrokerCustomerItem[];
    locations: string[];
    total: number;
  }>("/api/company-admin/brokers");

  brokers = safeArray<BrokerCustomerItem>(
    brokerResult.brokers,
  );
} catch (brokerError) {
  brokerLoadError =
    brokerError instanceof Error
      ? brokerError.message
      : "Could not load broker customers.";
}

setData({
  ...dashboardResult,
  brokers,
  brokerLoadError,
});
